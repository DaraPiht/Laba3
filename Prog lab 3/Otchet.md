**7\.** Для решение поставленной задачи пишем код:
```c++
#include <iostream>
#include "libbmp.h"

using namespace std;

int Bit = 7;  
int Byte = 0; 
char text[4000]; 
bool _end = false;

void pls_bit(char b) 
{
	if (_end)
		return; 

	b = b << Bit;
	text[Byte] = text[Byte] | b;
	Bit--;
	
	if (Bit < 0)
	{
		if (text[Byte] == '\0')
		{
			_end = true;
		}

		Bit = 7;
		Byte++;
	}
}

int main()
{

	BmpImg pic;
	pic.read("pic7.bmp");
	
	for (int x = 0; x <= pic.get_width() - 1; x++) 
	{
		for (int y = 0; y <= pic.get_height() - 1; y++)
		{

			char b1 = pic.red_at(x, y) & 0b1;
			char b2 = pic.blue_at(x, y) & 0b1;
			char b3 = pic.green_at(x, y) & 0b1;

			pls_bit(b1); pls_bit(b2); pls_bit(b3);

			if (_end)
				break;
		}

		if (_end)
			break;
	}

	cout << text<<"\n";
}
```
